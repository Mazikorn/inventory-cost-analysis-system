### 2.3 测试用例示例

#### 2.3.1 效期计算测试

```typescript
import { calculateExpiryStatus, getDaysUntilExpiry } from '../utils/expiry';

describe('Expiry Utils', () => {
  describe('calculateExpiryStatus', () => {
    it('should return "normal" when expiry date is more than 30 days away', () => {
      const futureDate = new Date();
      futureDate.setDate(futureDate.getDate() + 31);
      
      const result = calculateExpiryStatus(futureDate);
      
      expect(result).toBe('normal');
    });

    it('should return "warning" when expiry date is within 30 days', () => {
      const futureDate = new Date();
      futureDate.setDate(futureDate.getDate() + 15);
      
      const result = calculateExpiryStatus(futureDate);
      
      expect(result).toBe('warning');
    });

    it('should return "danger" when expiry date is within 7 days', () => {
      const futureDate = new Date();
      futureDate.setDate(futureDate.getDate() + 5);
      
      const result = calculateExpiryStatus(futureDate);
      
      expect(result).toBe('danger');
    });

    it('should return "expired" when expiry date has passed', () => {
      const pastDate = new Date();
      pastDate.setDate(pastDate.getDate() - 1);
      
      const result = calculateExpiryStatus(pastDate);
      
      expect(result).toBe('expired');
    });
  });

  describe('getDaysUntilExpiry', () => {
    it('should calculate correct days until expiry', () => {
      const futureDate = new Date();
      futureDate.setDate(futureDate.getDate() + 10);
      
      const result = getDaysUntilExpiry(futureDate);
      
      expect(result).toBe(10);
    });

    it('should return negative days for expired items', () => {
      const pastDate = new Date();
      pastDate.setDate(pastDate.getDate() - 5);
      
      const result = getDaysUntilExpiry(pastDate);
      
      expect(result).toBe(-5);
    });
  });
});
```

#### 2.3.3 表单验证测试

```typescript
import { validateInboundForm } from '../utils/validation';

describe('Inbound Form Validation', () => {
  it('should pass validation with all required fields', () => {
    const formData = {
      category: 'immunology',
      quantity: 10,
      unit: 'bottle',
      unitPrice: 580,
      supplier: '罗氏诊断',
    };
    
    const result = validateInboundForm(formData);
    
    expect(result.isValid).toBe(true);
    expect(result.errors).toEqual({});
  });

  it('should fail validation with missing required fields', () => {
    const formData = {
      category: '',
      quantity: null,
      unit: '',
      unitPrice: null,
      supplier: '',
    };
    
    const result = validateInboundForm(formData);
    
    expect(result.isValid).toBe(false);
    expect(result.errors.category).toBe('请选择分类');
    expect(result.errors.quantity).toBe('请输入数量');
    expect(result.errors.unit).toBe('请选择单位');
    expect(result.errors.unitPrice).toBe('请输入单价');
    expect(result.errors.supplier).toBe('请输入供应商');
  });

  it('should fail validation with negative quantity', () => {
    const formData = {
      category: 'immunology',
      quantity: -10,
      unit: 'bottle',
      unitPrice: 580,
      supplier: '罗氏诊断',
    };
    
    const result = validateInboundForm(formData);
    
    expect(result.isValid).toBe(false);
    expect(result.errors.quantity).toBe('数量不能为负数');
  });
});
```

#### 3.3.1 入库 API 测试

```typescript
import request from 'supertest';
import app from '../app';
import { setupTestDB, teardownTestDB } from './helpers/database';

describe('Inbound API', () => {
  beforeAll(async () => {
    await setupTestDB();
  });

  afterAll(async () => {
    await teardownTestDB();
  });

  describe('POST /api/inbound', () => {
    it('should create inbound record successfully', async () => {
      const inboundData = {
        category: 'immunology',
        specification: '6ml/瓶',
        quantity: 10,
        unit: 'bottle',
        unitPrice: 580,
        supplier: '罗氏诊断',
        productionBatchNo: 'ROCHE-20260403-001',
        expiryDate: '2026-12-31',
        storageLocation: 'A区-1号柜-3层',
      };

      const response = await request(app)
        .post('/api/inbound')
        .set('Authorization', `Bearer ${validToken}`)
        .send(inboundData)
        .expect(201);

      expect(response.body.success).toBe(true);
      expect(response.body.data.batchNo).toMatch(/^\d{8}-\d{3}$/);
      expect(response.body.data.totalPrice).toBe(5800);
    });

    it('should return 400 for invalid data', async () => {
      const invalidData = {
        category: '',
        quantity: -10,
      };

      const response = await request(app)
        .post('/api/inbound')
        .set('Authorization', `Bearer ${validToken}`)
        .send(invalidData)
        .expect(400);

      expect(response.body.success).toBe(false);
      expect(response.body.errors).toBeDefined();
    });

    it('should return 401 for unauthorized request', async () => {
      const response = await request(app)
        .post('/api/inbound')
        .send({})
        .expect(401);

      expect(response.body.success).toBe(false);
    });
  });

  describe('GET /api/inbound/:id', () => {
    it('should return inbound record by id', async () => {
      const response = await request(app)
        .get('/api/inbound/1')
        .set('Authorization', `Bearer ${validToken}`)
        .expect(200);

      expect(response.body.success).toBe(true);
      expect(response.body.data.id).toBe(1);
    });

    it('should return 404 for non-existent record', async () => {
      const response = await request(app)
        .get('/api/inbound/99999')
        .set('Authorization', `Bearer ${validToken}`)
        .expect(404);

      expect(response.body.success).toBe(false);
    });
  });
});
```

#### 4.3.1 入库流程测试

```typescript
import { test, expect } from '@playwright/test';

test.describe('Inbound Flow', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/login');
    await page.fill('[data-testid="email"]', 'test@example.com');
    await page.fill('[data-testid="password"]', 'password123');
    await page.click('[data-testid="submit"]');
    await page.waitForURL('/dashboard');
  });

  test('should complete inbound flow successfully', async ({ page }) => {
    await page.goto('/inventory');
    
    await page.click('[data-testid="new-inbound-btn"]');
    
    await expect(page.locator('.modal')).toBeVisible();
    
    await page.selectOption('[data-testid="category-select"]', 'immunology');
    await page.fill('[data-testid="specification-input"]', '6ml/瓶');
    await page.fill('[data-testid="quantity-input"]', '10');
    await page.selectOption('[data-testid="unit-select"]', 'bottle');
    await page.fill('[data-testid="unit-price-input"]', '580');
    await page.fill('[data-testid="supplier-input"]', '罗氏诊断');
    await page.fill('[data-testid="expiry-date-input"]', '2026-12-31');
    await page.fill('[data-testid="storage-location-input"]', 'A区-1号柜-3层');
    
    await page.click('[data-testid="submit-btn"]');
    
    await expect(page.locator('.toast.success')).toBeVisible();
    await expect(page.locator('.toast')).toContainText('入库成功');
    
    await expect(page.locator('.modal')).not.toBeVisible();
    
    await expect(page.locator('table')).toContainText('罗氏诊断');
  });

  test('should show validation errors for empty fields', async ({ page }) => {
    await page.goto('/inventory');
    
    await page.click('[data-testid="new-inbound-btn"]');
    
    await page.click('[data-testid="submit-btn"]');
    
    await expect(page.locator('.form-error')).toContainText('请选择分类');
    await expect(page.locator('.form-error')).toContainText('请输入数量');
    await expect(page.locator('.form-error')).toContainText('请选择单位');
  });

  test('should calculate total price automatically', async ({ page }) => {
    await page.goto('/inventory');
    
    await page.click('[data-testid="new-inbound-btn"]');
    
    await page.fill('[data-testid="quantity-input"]', '10');
    await page.fill('[data-testid="unit-price-input"]', '580');
    
    const totalPrice = await page.locator('[data-testid="total-price-input"]').inputValue();
    expect(totalPrice).toBe('5800.00');
  });
});
```

## 5. CI/CD 集成

### 5.1 GitHub Actions 配置

```yaml
name: Test

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main, develop]

jobs:
  unit-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'
      - run: npm ci
      - run: npm run test:unit -- --coverage
      - uses: codecov/codecov-action@v3
        with:
          files: ./coverage/lcov.info

  integration-test:
    runs-on: ubuntu-latest
    services:
      postgres:
        image: postgres:15
        env:
          POSTGRES_USER: test
          POSTGRES_PASSWORD: test
          POSTGRES_DB: coreone_test
        ports:
          - 5432:5432
      redis:
        image: redis:7
        ports:
          - 6379:6379
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'
      - run: npm ci
      - run: npm run db:migrate:test
      - run: npm run test:integration

  e2e-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'
      - run: npm ci
      - run: npx playwright install --with-deps
      - run: npm run build
      - run: npm run test:e2e
      - uses: actions/upload-artifact@v3
        if: always()
        with:
          name: playwright-report
          path: playwright-report/
          retention-days: 30
```
