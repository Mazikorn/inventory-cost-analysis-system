## 3. 测试数据生成脚本

### 3.1 入库记录生成脚本

```javascript
const faker = require('faker');
const fs = require('fs');

const categories = ['immunology', 'biochemistry', 'staining', 'accessories', 'other'];
const units = ['bottle', 'box', 'bag', 'piece', 'set', 'ml', 'g', 'slice'];
const suppliers = ['罗氏诊断', '碧云天生物', '索莱宝科技', '江苏世泰', '达科为生物'];

function generateBatchNo(date, seq) {
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const day = String(date.getDate()).padStart(2, '0');
  return `${year}${month}${day}-${String(seq).padStart(3, '0')}`;
}

function generateExpiryDate(baseDate, days) {
  const expiry = new Date(baseDate);
  expiry.setDate(expiry.getDate() + days);
  return expiry.toISOString().split('T')[0];
}

function generateInboundRecord(index, expiryDays) {
  const today = new Date();
  return {
    batchNo: generateBatchNo(today, index),
    category: categories[Math.floor(Math.random() * categories.length)],
    specification: `${Math.floor(Math.random() * 100) + 1}${faker.random.arrayElement(['ml/瓶', 'ml/套', 'T/盒', '×26mm'])}`,
    quantity: parseFloat((Math.random() * 100).toFixed(2)),
    unit: units[Math.floor(Math.random() * units.length)],
    unitPrice: parseFloat((Math.random() * 1000).toFixed(2)),
    supplier: suppliers[Math.floor(Math.random() * suppliers.length)],
    productionBatchNo: `${faker.random.alpha({count: 3}).toUpperCase()}-${today.getFullYear()}${String(today.getMonth() + 1).padStart(2, '0')}${String(today.getDate()).padStart(2, '0')}-${String(index).padStart(3, '0')}`,
    expiryDate: generateExpiryDate(today, expiryDays),
    storageLocation: `${faker.random.alpha({count: 1}).toUpperCase()}区-${Math.floor(Math.random() * 5) + 1}号柜-${Math.floor(Math.random() * 5) + 1}层`,
    remark: faker.lorem.sentence()
  };
}

function generateTestData() {
  const records = [];
  let index = 1;

  for (let i = 0; i < 20; i++) {
    records.push(generateInboundRecord(index++, Math.floor(Math.random() * 300) + 31));
  }

  for (let i = 0; i < 15; i++) {
    records.push(generateInboundRecord(index++, Math.floor(Math.random() * 23) + 8));
  }

  for (let i = 0; i < 10; i++) {
    records.push(generateInboundRecord(index++, Math.floor(Math.random() * 7) + 1));
  }

  for (let i = 0; i < 5; i++) {
    records.push(generateInboundRecord(index++, -Math.floor(Math.random() * 30) - 1));
  }

  return records;
}

const testData = generateTestData();
fs.writeFileSync('test-data-inbound.json', JSON.stringify(testData, null, 2));
console.log(`Generated ${testData.length} test records`);
```

### 3.2 性能测试数据生成脚本

function generatePerformanceTestData(count) {
  const records = [];
  const today = new Date();
  
  for (let i = 1; i <= count; i++) {
    records.push(generateInboundRecord(i, Math.floor(Math.random() * 365) + 1));
  }
  
  return records;
}

const performanceData = generatePerformanceTestData(10000);
fs.writeFileSync('test-data-performance.json', JSON.stringify(performanceData, null, 2));
console.log(`Generated ${performanceData.length} performance test records`);

## 4. 测试数据管理

### 4.1 数据存储

| 数据类型 | 存储位置 | 格式 | 更新频率 |
|---------|---------|------|----------|
| 正常数据 | `/test-data/normal/` | JSON | 每次测试前 |
| 边界数据 | `/test-data/boundary/` | JSON | 每次测试前 |
| 异常数据 | `/test-data/exception/` | JSON | 每次测试前 |
| 性能数据 | `/test-data/performance/` | JSON | 性能测试前 |

### 4.2 数据清理

- **测试前**: 清空测试数据库，导入初始测试数据
- **测试后**: 清空测试数据库，避免污染环境
- **定期清理**: 每周清理一次测试数据目录

### 4.3 数据隐私

- **不使用真实患者数据**: 所有测试数据均为虚构
- **不使用真实供应商数据**: 供应商名称为示例数据
- **不使用真实价格数据**: 价格为随机生成，仅供参考
